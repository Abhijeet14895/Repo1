step 1: check if git is already installed 
	>git --version
step 2: download and install git
step 3: add your project to git
	create a folder/tree structure and navigate to the location using cmd
	>git init # this will add git binary file to the project/folder and start tracking
step 4: navigate to github and signup (create your github account)
	open cmd and enter below commands to refer/map your github account repo
	>git config --global user.email "abhijeet.dhage@live.in"
	>git config --global user.name "Abhijeet14895"
step 5: basic commands
	>git status
	>dir > test1.txt # creating new file using command line
	>git add # create new test1.txt file in folder 
	>git commit -m "your message"
	>git remote add origin "https://github.com/Abhijeet14895/Repo1.git" # Sign in to github, create repository, map with local repo
	>git push -u origin main
	>git log
	>git --help
step 6: Create branch
	>git branch <branch name>
step 7: Checkout branch
	>git checkout <branch name>
step 8: merge new branch in main
	>git checkout main # ensure you are in main/master branch
	>git merge <branch name> # merge create a separate merge commit on current branch having latest changes from main
step 9: rebase new branch with main branch latest changes
	>git checkout <branch name>
	>git rebase main # rebase changes the base to latest changes from main
step 9: delete branch
	>git checkout main # ensure you are in main/master branch
	>git branch -d "branch name" # delete from local
	>git push origin --delete "branch name" # delete from remote
step 10: add tag
	>git checkout <branch name>
	>git tag <tag name>
	>git tag v1.0 OR git tag -a v1.0 -m "your message" # use 2nd one to create annotated tags
	>git tag OR git tag -l "v1*" # use 2nd one to filter search"
	>git show v1.0 # gives tag details
	>git push origin v1.0 # pushing specific tag to remote
	>git push origin --tags # pushing all tags to remote
	>git tag --delete v1.0 # deleting specific tag
	>git push origin --delete v1.0 # pushing to remote
	>git tag -d v1.0 v1.1 # deleting multiple tags
	>git push origin -d v1.0 v1.1 # pushing to remote
	>git checkout -b <branch name> <tag Name> # we can create branch from tag and checkout branch
	>git tag <tag name> <reference of commit> # how to create a tag from past commit
